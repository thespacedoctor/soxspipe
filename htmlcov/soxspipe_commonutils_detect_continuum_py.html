98 def get(self): 

99 """ 

100 *return the order centre table filepath* 

101 

102 **Return:** 

103 - ``order_table_path`` -- file path to the order centre table giving polynomial coeffs to each order fit 

104 """ 

105 self.log.debug('starting the ``get`` method') 

106 

107 arm = self.arm 

108 

109 # READ THE SPECTRAL FORMAT TABLE TO DETERMINE THE LIMITS OF THE TRACES 

110 orderNums, waveLengthMin, waveLengthMax = self.read_spectral_format() 

111 

112 # CONVERT WAVELENGTH TO PIXEL POSTIONS AND RETURN ARRAY OF POSITIONS TO 

113 # SAMPLE THE TRACES 

114 pixelArrays = self.create_pixel_arrays( 

115 orderNums, 

116 waveLengthMin, 

117 waveLengthMax) 

118 

119 # SLICE LENGTH TO SAMPLE TRACES IN THE CROSS-DISPERSION DIRECTION 

120 sliceLength = self.settings[

123 "soxs-order-centre"]["peak-sigma-limit"]

277 def fit_1d_gaussian_to_slice( 

278 self, 

279 xy, 

280 sliceLength, 

281 peakSigmaLimit): 

282 """*cut a slice from the pinhole flat along the cross-dispersion direction centred on pixel position, fit 1D gaussian and return the peak pixel position* 

283 

284 **Key Arguments:** 

285 - ``xy`` -- the x,y pixel coordinate (tuple) 

286 - ``sliceLength`` -- length of the slice to cut 

287 - ``peakSigmaLimit`` -- number of simga above median. If peak is not above this level slice will be rejected 

288 

289 **Return:** 

290 - ``fit_xy`` -- gaussian fit peak xy position 

291 """ 

292 self.log.debug('starting the ``fit_1d_gaussian_to_slice`` method') 

293 

294 # CLIP OUT A SLICE TO INSPECT CENTRED AT POSITION 

295 halfSlice = sliceLength / 2 

296 x_fit = xy[0] 

297 y_fit = xy[1] 

298 

299 try: 

300 slice = self.pinholeFlat.data[int(y_fit), max( 

301 0, int(x_fit - halfSlice)):min(2048, int(x_fit + halfSlice))] 

302 except: 

303 return None 

304 

305 # CHECK THE SLICE POINTS IF NEEDED 

306 if 1 == 0: 

307 x = np.arange(0, len(slice)) 

308 plt.figure(figsize=(8, 5)) 

309 plt.plot(x, slice, 'ko') 

310 plt.xlabel('Position') 

311 plt.ylabel('Flux') 

312 plt.show() 

313 

314 # EVALUATING THE MEAN AND STD-DEV FOR PEAK FINDING - REMOVES SLICE 

315 # CONTAINING JUST NOISE 

316 median_r = np.median(slice) 

317 std_r = mad_std(slice) 

318 peaks, _ = find_peaks(slice, height=median_r + 

319 peakSigmaLimit * std_r, width=1) 

320 

321 # CHECK PEAK HAS BEEN FOUND 

322 if peaks is None or len(peaks) <= 0: 

323 # CHECK THE SLICE POINTS IF NEEDED 

324 if 1 == 0: 

325 x = np.arange(0, len(slice)) 

326 plt.figure(figsize=(8, 5)) 

327 plt.plot(x, slice, 'ko') 

328 plt.xlabel('Position') 

329 plt.ylabel('Flux') 

330 plt.show() 

331 return None 

332 

333 # FIT THE DATA USING A 1D GAUSSIAN - USING astropy.modeling 

334 # CENTRE THE GAUSSIAN ON THE PEAK 

335 g_init = models.Gaussian1D( 

336 amplitude=1000., mean=peaks[0], stddev=1.) 

337 # print(f"g_init: {g_init}") 

338 fit_g = fitting.LevMarLSQFitter() 

339 

340 # NOW FIT 

341 g = fit_g(g_init, np.arange(0, len(slice)), slice) 

342 xpeak = g.mean + max(0, int(x_fit - halfSlice)) 

343 ypeak = y_fit 

344 

345 # PRINT A FEW PLOTS IF NEEDED - GUASSIAN FIT OVERLAYED 

346 if 1 == 0 and random() < 0.02: 

347 x = np.arange(0, len(slice)) 

348 plt.figure(figsize=(8, 5)) 

349 plt.plot(x, slice, 'ko') 

350 plt.xlabel('Position') 

351 plt.ylabel('Flux') 

352 guassx = np.arange(0, max(x), 0.05) 

353 plt.plot(guassx, g(guassx), label='Gaussian') 

354 plt.show() 

355 

356 self.log.debug('completed the ``fit_1d_gaussian_to_slice`` method') 

357 return (xpeak, ypeak)

359 def fit_polynomial( 

360 self, 

361 pixelArray): 

362 """*iteratively fit the dispersion map polynomials to the data, clipping residuals with each iteration* 

363 

364 **Key Arguments:** 

365 - ``pixelArray`` -- the array of x,y pixels of measured 1D guassian peak positions 

366 

367 **Return:** 

368 - ``coeffs`` -- the coefficients of the polynomial fit 

369 - ``residuals`` -- the residuals of the fit compared to original guassian peak positions 

370 - ``xfit`` -- the polynomial fits x-positions 

371 - ``xcoords`` -- the clean guassian peak x-coordinate list (post-clipping) 

372 - ``ycoords`` -- the clean guassian peak x-coordinate list (post-clipping) 

373 """ 

374 self.log.debug('starting the ``fit_polynomial`` method') 

375 

376 arm = self.arm 

377 

378 clippedCount = 1 

379 

380 poly = chebyshev_xy_polynomial( 

381 log=self.log, deg=self.polyDeg).poly 

382 

383 clippingSigma = self.settings[ 

384 "soxs-order-centre"]["poly-fitting-residual-clipping-sigma"] 

385 

386 clippingIterationLimit = self.settings[ 

387 "soxs-order-centre"]["clipping-iteration-limit"] 

388 

389 xcoords = [v[0] for v in pixelArray] 

390 ycoords = [v[1] for v in pixelArray] 

391 

392 iteration = 0 

393 while clippedCount > 0 and iteration < clippingIterationLimit: 

394 iteration += 1 

395 # USE LEAST-SQUARED CURVE FIT TO FIT CHEBY POLY 

396 coeff = np.ones((self.polyDeg + 1)) 

397 coeff, pcov_x = curve_fit( 

398 poly, xdata=ycoords, ydata=xcoords, p0=coeff) 

399 

400 residuals, mean_res, std_res, median_res, xfit = self.calculate_residuals( 

401 xcoords=xcoords, 

402 ycoords=ycoords, 

403 coeff=coeff) 

404 

405 # SIGMA-CLIP THE DATA 

406 masked_residuals = sigma_clip( 

407 residuals, sigma_lower=clippingSigma, sigma_upper=clippingSigma, maxiters=1, cenfunc='median', stdfunc=mad_std) 

408 

409 # MASK DATA ARRAYS WITH CLIPPED RESIDUAL MASK 

410 startCount = len(xcoords) 

411 a = [xcoords, ycoords, residuals] 

412 xcoords, ycoords, residuals = [np.ma.compressed(np.ma.masked_array( 

413 i, masked_residuals.mask)) for i in a] 

414 clippedCount = startCount - len(xcoords) 

415 print(f'{clippedCount} pixel positions where clipped in this iteration of fitting an order centre polynomial') 

416 

417 self.log.debug('completed the ``fit_polynomials`` method') 

418 return coeff, residuals, xfit, xcoords, ycoords

420 def calculate_residuals( 

421 self, 

422 xcoords, 

423 ycoords, 

424 coeff): 

425 """*calculate residuals of the polynomial fits against the observed line postions* 

426 

427 **Key Arguments:** 

428 - ``xcoords`` -- the measured x positions of the gaussian peaks 

429 - ``ycoords`` -- the measurd y positions of the gaussian peaks 

430 - ``coeff`` -- the coefficients of the fitted polynomial 

431 

432 **Return:** 

433 - ``residuals`` -- x residuals 

434 - ``mean`` -- the mean of the residuals 

435 - ``std`` -- the stdev of the residuals 

436 - ``median`` -- the median of the residuals 

437 """ 

438 self.log.debug('starting the ``calculate_residuals`` method') 

439 

440 arm = self.arm 

441 

442 poly = chebyshev_xy_polynomial( 

443 log=self.log, deg=self.polyDeg).poly 

444 

445 # CALCULATE RESIDUALS BETWEEN GAUSSIAN PEAK LINE POSITIONS AND POLY 

446 # FITTED POSITIONS 

447 xfit = poly( 

448 ycoords, *coeff) 

449 res_x = np.asarray(xfit) - np.asarray(xcoords) 

450 

451 # CALCULATE COMBINED RESIDUALS AND STATS 

452 res_mean = np.mean(res_x) 

453 res_std = np.std(res_x) 

454 res_median = np.median(res_x) 

455 

456 self.log.debug('completed the ``calculate_residuals`` method') 

457 return res_x, res_mean, res_std, res_median, xfit

562 def write_order_table_to_file( 

563 self, 

564 orderLoctions): 

565 """*write out the fitted polynomial solution coefficients to file* 

566 

567 **Key Arguments:** 

568 - ``orderLoctions`` -- dictionary of the order coefficients 

569 

570 **Return:** 

571 - ``order_table_path`` -- path to the order table file 

572 """ 

573 self.log.debug('starting the ``write_order_table_to_file`` method') 

574 

575 arm = self.arm 

576 

577 # SORT COEFFICIENT OUTPUT TO WRITE TO FILE 

578 listOfDictionaries = [] 

579 for k, v in orderLoctions.items(): 

580 orderDict = collections.OrderedDict(sorted({}.items())) 

581 orderDict["order"] = int(k.replace("o", "")) 

582 orderDict["degy"] = self.polyDeg 

583 n_coeff = 0 

584 for i in range(0, self.polyDeg + 1): 

585 orderDict[f'CENT_c{i}'] = v[n_coeff] 

586 n_coeff += 1
