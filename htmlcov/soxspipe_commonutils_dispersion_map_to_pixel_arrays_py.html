27 """*use a first-guess dispersion map to append x,y fits to line-list data frame.*  

28 

29 Return a line-list with x,y fits given a first guess dispersion map.* 

30 

31 **Key Arguments:** 

32 

33 - `log` -- logger 

34 - `dispersionMapPath` -- path to the dispersion map 

35 - `lineList` -- a data-frame including 'Order', 'Wavelength' and 'slit_pos' columns 

36 

37 **Usage:** 

38 

39 ```python 

40 myDict = { 

41 "Order": [11, 11, 11], 

42 "Wavelength": [850.3, 894.3, 983.2], 

43 "slit_pos": [0, 0, 0] 

44 } 

45 lineList = pd.DataFrame(myDict) 

46 lineList = dispersion_map_to_pixel_arrays( 

47 log=log, 

48 dispersionMapPath="/path/to/map.csv", 

49 lineList=lineList 

50 ) 

51 ```  

52 """ 

53 log.debug('starting the ``dispersion_map_to_pixel_arrays`` function') 

54 

55 # READ THE FILE 

56 home = expanduser("~") 

57 dispersion_map = dispersionMapPath.replace("~", home) 

58 

59 # READ IN THE X- AND Y- GENERATING POLYNOMIALS FROM DISPERSION MAP FILE 

60 coeff = {} 

61 poly = {} 

62 with open(dispersion_map, 'rb') as csvFile: 

63 csvReader = csv.DictReader( 

64 csvFile, dialect='excel', delimiter=',', quotechar='"') 

65 for row in csvReader: 

66 axis = row["axis"] 

67 order_deg = int(row["order-deg"]) 

68 wavelength_deg = int(row["wavelength-deg"]) 

69 slit_deg = int(row["slit-deg"]) 

70 coeff[axis] = [float(v) for k, v in row.items() if k not in [ 

71 "axis", "order-deg", "wavelength-deg", "slit-deg"]] 

72 poly[axis] = chebyshev_order_wavelength_polynomials( 

73 log=log, order_deg=order_deg, wavelength_deg=wavelength_deg, slit_deg=slit_deg).poly 

74 csvFile.close() 

75 

76 # CONVERT THE ORDER-SORTED WAVELENGTH ARRAYS INTO ARRAYS OF PIXEL TUPLES 

77 lineList["fit_x"] = poly['x'](lineList, *coeff['x']) 

78 lineList["fit_y"] = poly['y'](lineList, *coeff['y']) 

79 

80 # FILTER DATA FRAME 

81 # FIRST CREATE THE MASK 

82 mask = (lineList["fit_x"] > 0) & (lineList["fit_y"] > 0) 

83 lineList = lineList.loc[mask] 

84 

85 log.debug('completed the ``dispersion_map_to_pixel_arrays`` function') 

86 return lineList
